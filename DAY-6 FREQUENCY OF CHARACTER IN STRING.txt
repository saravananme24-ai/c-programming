#include <stdio.h>
#include <string.h> 

int main() {
    char str[1000]; 
    char target_char; 
    int count = 0;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 

    printf("Enter the character to find its frequency: ");
    scanf(" %c", &target_char); 
    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == target_char) {
            count++; 
        }
    }

    printf("Frequency of '%c' in the string is: %d\n", target_char, count);

    return 0;
}
