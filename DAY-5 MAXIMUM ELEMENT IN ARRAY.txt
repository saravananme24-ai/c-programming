#include <stdio.h>
int main() {
   int n;
   printf("Enter the size of the array: ");
   scanf("%d", &n);
   int arr[n];
   printf("Enter %d space-separated integers: ", n);
   for (int i = 0; i < n; i++) {
       scanf("%d", &arr[i]);
   }
   printf("Input Array: ");
   for (int i = 0; i < n; i++) {
       printf("%d ", arr[i]);
   }
   printf("\n");
   int max = arr[0];
   for (int i = 1; i < n; i++) {
       if (arr[i] > max) {
           max = arr[i];
       }
   }
   printf("Maximum Element: %d\n", max);
   return 0;
}
