#include <stdio.h>
#include <string.h> 
int main() {
    char str[100]; 
    int count = 0; 
    int i;

    printf("Enter a string: ");
    
    fgets(str, sizeof(str), stdin);

    str[strcspn(str, "\n")] = 0; 

    if (strlen(str) == 0) {
        printf("Number of words: 0\n");
        return 0;
    }

    for (i = 0; str[i] != '\0'; i++) {
        
        if (str[i] == ' ' || str[i] == '\n' || str[i] == '\t') {
            
            if (i > 0 && !(str[i-1] == ' ' || str[i-1] == '\n' || str[i-1] == '\t')) {
                count++;
            }
        }
    }

    
    if (strlen(str) > 0) {
        count++;
    }

    printf("Number of words: %d\n", count);

    return 0;
}
