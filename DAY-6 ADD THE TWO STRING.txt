 #include <stdio.h>

void concatenateStrings(char *s1, const char *s2) {
    int i = 0;
    
    while (s1[i] != '\0') {
        i++;
    }

    int j = 0;
    
    while (s2[j] != '\0') {
        s1[i] = s2[j];
        i++;
        j++;
    }
   
    s1[i] = '\0';
}

int main() {
    char string1[100] = "Hello ";
    char string2[] = "World!";

    concatenateStrings(string1, string2);

    printf("Concatenated string: %s\n", string1);

    return 0;
}
